Add the following JavaScript inside the existing `$(document).ready(function() { ... })` block, after the database reset handler (after line 300):

```javascript
// --- Database Export ---
$('#export-db-btn').on('click', function() {
    $(this).prop('disabled', true).text('Wird heruntergeladen...');
    // Use a hidden link to trigger download without page reload
    const link = document.createElement('a');
    link.href = '/api/settings/db/export';
    link.style.display = 'none';
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    // Re-enable after short delay (download starts async)
    setTimeout(() => {
        $('#export-db-btn').prop('disabled', false).text('Datenbank herunterladen');
    }, 2000);
});

// --- Database Import ---
$('#import-db-file').on('change', function() {
    const file = this.files[0];
    $('#import-db-error').hide();
    $('#import-db-success').hide();
    if (file && file.name.endsWith('.db')) {
        $('#import-db-btn').prop('disabled', false);
    } else {
        $('#import-db-btn').prop('disabled', true);
        if (file) {
            $('#import-db-error').text('Nur .db-Dateien sind erlaubt.').show();
        }
    }
});

$('#import-db-btn').on('click', function() {
    if (!confirm('Achtung: Die bestehende Datenbank wird ueberschrieben. Moechten Sie fortfahren?')) {
        return;
    }
    const file = $('#import-db-file')[0].files[0];
    if (!file) return;

    const formData = new FormData();
    formData.append('db_file', file);

    $(this).prop('disabled', true).text('Wird importiert...');
    $('#import-db-error').hide();
    $('#import-db-success').hide();

    $.ajax({
        url: '/api/settings/db/import',
        method: 'POST',
        data: formData,
        processData: false,
        contentType: false,
        headers: { 'X-CSRFToken': $('meta[name="csrf-token"]').attr('content') },
        success: function(response) {
            if (response.success) {
                $('#import-db-success').text(response.message).show();
                showAlert(response.message, 'success');
                // Reload page after short delay to reflect new DB
                setTimeout(() => window.location.reload(), 2000);
            } else {
                $('#import-db-error').text(response.message).show();
            }
        },
        error: function(xhr) {
            const errorMsg = xhr.responseJSON ? xhr.responseJSON.message : 'Ein Serverfehler ist aufgetreten.';
            $('#import-db-error').text(errorMsg).show();
        },
        complete: function() {
            $('#import-db-btn').prop('disabled', false).text('Datenbank importieren');
            $('#import-db-file').val('');
        }
    });
});
```
