## Create the Final, Production-Grade systemd Service File
##########################################
sudo nano /etc/systemd/system/stecher-tennis.service

[Unit]
Description=Gunicorn instance for Stecher Tennis App
After=network.target

[Service]
User=stecher
Group=www-data
WorkingDirectory=/home/stecher/stecher_tennis
EnvironmentFile=/home/stecher/stecher_tennis/.env

# --- DEFINITIVE FIX: Use a SINGLE eventlet worker ---
# A single, async worker is more efficient on a Raspberry Pi and
# completely solves the "Invalid session" problem by ensuring all
# connections are handled by the same process.
ExecStart=/home/stecher/stecher_tennis/bin/gunicorn --workers 1 --worker-class eventlet --bind 127.0.0.1:8000 --timeout 120 app:app

Restart=always
RestartSec=10
StandardOutput=journal
StandardError=journal
SyslogIdentifier=stecher-tennis
MemoryMax=2G
CPUQuota=80%
LimitNOFILE=65536
PrivateTmp=true
ProtectSystem=full
NoNewPrivileges=true
ProtectHome=false

[Install]
WantedBy=multi-user.target

### activate unattended upgrades

*) sudo apt update
   sudo apt install unattended-upgrades apt-listchanges
   sudo dpkg-reconfigure --priority=low unattended-upgrades   # enables 20auto-upgrades
   
   ---> dpkg-reconfigure writes /etc/apt/apt.conf.d/20auto-upgrades with
   
        APT::Periodic::Update-Package-Lists "1";
        APT::Periodic::Unattended-Upgrade "1";
        
        and leaves the heavy lifting to two timers that already exist:
        apt-daily.timer (metadata refresh) and apt-daily-upgrade.timer (unattended-upgrade run)

   *) Verify timers are active
      systemctl list-timers --all | grep apt

   *) Pin the upgrade to 03:30 Sunday so it finishes before your 04:00 reboot
      sudo mkdir -p /etc/systemd/system/apt-daily-upgrade.timer.d

      sudoedit /etc/systemd/system/apt-daily-upgrade.timer.d/override.conf
      
      [Timer]
      OnCalendar=
      OnCalendar=Sun *-*-* 03:30:00
      RandomizedDelaySec=0

   *) Reload & check:
      sudo systemctl daemon-reload
      sudo systemctl restart apt-daily-upgrade.timer
      systemctl list-timers apt-daily-upgrade.timer

   *) Tune unattended-upgrades
      sudo nano /etc/apt/apt.conf.d/50unattended-upgrades
      
      // Remove unused automatically installed kernel-related packages
      Unattended-Upgrade::Remove-Unused-Kernel-Packages "true";
      
      // Do automatic removal of newly unused dependencies after the upgrade
      Unattended-Upgrade::Remove-New-Unused-Dependencies "true";
      
      // Don’t let it auto-reboot; you already reboot every week
      Unattended-Upgrade::Automatic-Reboot "false";

   *) Inspect logs & test

PAM = Pluggable Authentication Modules
UsePAM	no
This is a significant change with potential side effects. PAM (Pluggable Authentication Modules)
is used by the system for more than just authentication. It also handles account management
(e.g., checking for expired accounts) and session management (e.g., setting resource limits or
displaying the message of the day). By setting UsePAM no, you are disabling these system-level
checks for SSH sessions. Since you have already disabled password authentication, the security
benefit of disabling PAM is minimal, but the risk of unintended consequences is higher.
---> Recommendation: It is generally safer and recommended to leave this enabled (UsePAM yes).
     Your server remains secure because PasswordAuthentication no and AuthenticationMethods publickey
