## Create the Final, Production-Grade systemd Service File
##########################################
sudo nano /etc/systemd/system/stecher-tennis.service

[Unit]
Description=Gunicorn instance for Stecher Tennis App
# This service will only start after the network is available
After=network.target

[Service]
# Run the service as the 'stecher' user and 'www-data' group
# (Ensure 'stecher' is in the 'www-data' group: sudo usermod -aG www-data stecher)
User=stecher
Group=www-data

# Set the working directory to the project folder
WorkingDirectory=/home/stecher/stecher_tennis

# Load secrets and configuration from a secure file
EnvironmentFile=/home/stecher/stecher_tennis/.env

# Command to start the Gunicorn server with Eventlet
# - Uses absolute path to the gunicorn in the virtual environment
# - Sets a stable number of workers suitable for a Raspberry Pi 4
# - Binds to localhost on port 8000 (for the reverse proxy)
ExecStart=/home/stecher/stecher_tennis/bin/gunicorn --workers 4 --worker-class eventlet --bind 127.0.0.1:8000 --timeout 120 app:app

# Always restart the service if it fails
Restart=always
RestartSec=10

# Logging configuration
StandardOutput=journal
StandardError=journal
SyslogIdentifier=stecher-tennis

# Resource limits for stability
MemoryMax=2G
CPUQuota=80%
LimitNOFILE=65536

# Security hardening directives
PrivateTmp=true
ProtectSystem=full
NoNewPrivileges=true
# ProtectHome is set to false because the app runs from a home directory.
# If the app were in /srv/ or /var/www/, this could be set to 'read-only' or 'true'.
ProtectHome=false

[Install]
# Enable the service to start automatically on boot
WantedBy=multi-user.target

# Caddyfile for the Tennis Ranking App
tc-breakpoint-forderung.duckdns.org {
    # Enable automatic HTTPS with Let's Encrypt
    # The email is used for certificate renewal notices.
    tls otto.kuegerl@gmail.com

    # Set up logging for this site
    # This is very useful for troubleshooting.
    log {
        output file /var/log/caddy/tc-breakpoint.log
        level ERROR
    }

    # Reverse proxy all requests to the Gunicorn application
    # running on localhost port 8000.
    reverse_proxy localhost:8000 {
        # Best Practice: Forward the original host and client IP headers
        # This allows your Flask app to know the real domain and user IP.
        header_up Host {http.request.host}
        header_up X-Real-IP {http.request.remote}
    }
}
